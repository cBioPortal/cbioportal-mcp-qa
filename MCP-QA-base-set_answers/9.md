total_samples AS (
SELECT COUNT(DISTINCT g.sample_unique_id) as total
FROM genomic_event_derived g
JOIN clinical_data_derived c ON g.sample_unique_id = c.sample_unique_id
WHERE g.hugo_gene_symbol = 'KRAS'
AND g.variant_type = 'mutation'
AND c.attribute_name = 'CANCER_TYPE'

SELECT
km.mutation_variant,
km.mutation_type,
km.sample_count,
ROUND((km.sample_count * 100.0) / ts.total, 2) as percentage
