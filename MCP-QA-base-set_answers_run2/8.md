### Query 5
**Timestamp:** 2025-09-30 18:05:29

```sql
SELECT
cs.cancer_study_identifier,
cs.name,
COUNT(DISTINCT ged.patient_unique_id) as patients_with_egfr_amplification,
COUNT(DISTINCT p.internal_id) as total_patients,
ROUND(COUNT(DISTINCT ged.patient_unique_id) * 100.0 / COUNT(DISTINCT p.internal_id), 1) as percentage
FROM cancer_study cs
JOIN patient p ON cs.cancer_study_id = p.cancer_study_id
LEFT JOIN genomic_event_derived ged ON ged.cancer_study_identifier = cs.cancer_study_identifier
AND ged.hugo_gene_symbol = 'EGFR'
AND ged.variant_type = 'cna'
AND ged.cna_alteration = 2
WHERE cs.cancer_study_identifier IN ('luad_tcga_gdc', 'luad_tcga_pub', 'luad_tcga_pan_can_atlas_2018', 'luad_tcga')
GROUP BY cs.cancer_study_identifier, cs.name
ORDER BY cs.cancer_study_identifier
```
