WITH mutation_counts AS (

    WHERE cancer_study_identifier = 'target_os_target_2018_pub'
)
SELECT 
    m.hugo_gene_symbol,
    m.mutated_samples,
    t.total_samples,
    ROUND((m.mutated_samples * 100.0 / t.total_samples), 1) as mutation_frequency_percent
FROM mutation_counts m
CROSS JOIN total_samples t
ORDER BY m.mutated_samples DESC
LIMIT 5
