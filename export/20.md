SELECT
cd_sex.attribute_value as sex,
cd_cancer.attribute_value as cancer_type,
COUNT(DISTINCT cd_sex.patient_unique_id) as patient_count,
AVG(CAST(cd_mutation.attribute_value AS Float64)) as avg_mutation_count
FROM cgds_public_2025_06_24.clinical_data_derived cd_sex
JOIN cgds_public_2025_06_24.clinical_data_derived cd_cancer
ON cd_sex.patient_unique_id = cd_cancer.patient_unique_id
JOIN cgds_public_2025_06_24.clinical_data_derived cd_mutation
ON cd_sex.patient_unique_id = cd_mutation.patient_unique_id
WHERE cd_sex.cancer_study_identifier = 'msk_chord_2024'
AND cd_cancer.cancer_study_identifier = 'msk_chord_2024'
AND cd_mutation.cancer_study_identifier = 'msk_chord_2024'
AND cd_sex.attribute_name = 'SEX'
AND cd_cancer.attribute_name = 'CANCER_TYPE_DETAILED'
AND cd_mutation.attribute_name = 'MUTATION_COUNT'
AND cd_cancer.attribute_value LIKE '%Large Cell%Lung%'
AND cd_sex.attribute_value IN ('Male', 'Female')
AND cd_mutation.attribute_value != ''
AND cd_mutation.attribute_value != 'NA'
GROUP BY cd_sex.attribute_value, cd_cancer.attribute_value
ORDER BY avg_mutation_count DESC
