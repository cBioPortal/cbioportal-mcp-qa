```sql
WITH tp53_mutated_samples AS (
SELECT DISTINCT sample_unique_id
FROM cgds_public_2025_06_24.genomic_event_derived
WHERE cancer_study_identifier = 'msk_chord_2024'
AND hugo_gene_symbol = 'TP53'
AND variant_type = 'mutation'
),
lung_adenocarcinoma_samples AS (
SELECT DISTINCT sample_unique_id
FROM cgds_public_2025_06_24.clinical_data_derived
WHERE cancer_study_identifier = 'msk_chord_2024'
AND attribute_name = 'CANCER_TYPE_DETAILED'
AND lower(attribute_value) LIKE '%lung adenocarcinoma%'
),
tmb_scores AS (
SELECT
cd.sample_unique_id,
toFloat64(cd.attribute_value) as tmb_score
FROM cgds_public_2025_06_24.clinical_data_derived cd
INNER JOIN tp53_mutated_samples tp53 ON cd.sample_unique_id = tp53.sample_unique_id
INNER JOIN lung_adenocarcinoma_samples lac ON cd.sample_unique_id = lac.sample_unique_id
WHERE cd.cancer_study_identifier = 'msk_chord_2024'
AND cd.attribute_name = 'TMB_NONSYNONYMOUS'
AND cd.attribute_value != ''
AND cd.attribute_value IS NOT NULL
)
SELECT
count(*) as sample_count,
median(tmb_score) as median_tmb,
min(tmb_score) as min_tmb,
max(tmb_score) as max_tmb,
avg(tmb_score) as mean_tmb
FROM tmb_scores
```
