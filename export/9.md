### Query 1
**Timestamp:** 2025-07-22 18:14:14

```sql
SELECT
hugo_gene_symbol,
COUNT(DISTINCT sample_unique_id) as mutated_samples,
COUNT(DISTINCT sample_unique_id) * 100.0 / (
SELECT COUNT(DISTINCT sample_unique_id)
FROM cgds_public_2025_06_24.clinical_data_derived cdd
WHERE cdd.cancer_study_identifier = 'msk_chord_2024'
AND cdd.attribute_name = 'CANCER_TYPE'
AND cdd.attribute_value = 'Non-Small Cell Lung Cancer'
) as mutation_frequency_percent
FROM cgds_public_2025_06_24.genomic_event_derived ged
WHERE ged.cancer_study_identifier = 'msk_chord_2024'
AND ged.variant_type = 'mutation'
AND ged.sample_unique_id IN (
SELECT DISTINCT sample_unique_id
FROM cgds_public_2025_06_24.clinical_data_derived
WHERE cancer_study_identifier = 'msk_chord_2024'
AND attribute_name = 'CANCER_TYPE'
AND attribute_value = 'Non-Small Cell Lung Cancer'
)
GROUP BY hugo_gene_symbol
ORDER BY mutated_samples DESC
LIMIT 20
```
