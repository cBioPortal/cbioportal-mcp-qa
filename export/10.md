liver_status.attribute_value as liver_involvement,

AND attribute_name = 'CANCER_TYPE'
AND attribute_value = 'Breast Cancer'
) breast_patients
JOIN cgds_public_2025_06_24.clinical_data_derived liver_status
ON breast_patients.patient_unique_id = liver_status.patient_unique_id
AND liver_status.cancer_study_identifier = 'msk_chord_2024'
AND liver_status.attribute_name = 'LIVER'
GROUP BY liver_status.attribute_value
